<script>

// 获取 Canvas 元素

var canvas = document.getElementById("myCanvas");

var ctx = canvas.getContext("2d");

  

// 经纬度坐标（你提供的坐标点）

var coordinates = [

  [31.827888602723082, 118.78522345854482],

  [31.82811559836081, 118.78512972252483],

  

  [31.828115657802382, 118.78512971308447],

  [31.828654241633203, 118.78490730868499],

  [31.82866753867539, 118.7849679723297],

  
  

  [31.828670361055824, 118.78498093921384],

  [31.8286666 ,118.7849831],

  [31.8286676 , 118.7849825],

  [ 31.8286688 , 118.7849818],

  [31.8286694 , 118.7849814],

[31.8286651 , 118.7849831],

  [31.8286556 , 118.7849871],

  [31.8286496 , 118.7849896],

  [31.827909606883516, 118.78529508590593],

  
  
  

  // 断点3  

  // [31.827909606883516, 118.78529508590593],

  
  

  // [31.828686564052447, 118.78505455460126],

  // [31.82793048788508, 118.78536676894119],

  // [31.827951478400685, 118.78543841147523],

  // [31.82870279150358, 118.78512816460389],

  // [31.828719018911137, 118.78520177463032],

  // [31.827972468874655, 118.78551005403993]

];

  

// 将经纬度转换为 Canvas 上的坐标

function convertLatLonToPixel(lat, lon) {

  // 这里假设 Canvas 大小为 500x500，可根据实际情况调整

  var x = (lon - 118.7849) * (canvas.width / 0.001) + 100; // 将经度范围缩小到 0.003 这样的区间

  var y = canvas.height - ((lat - 31.828) * (canvas.height / 0.001))  - 100; // 将纬度范围缩小到 0.003 这样的区间

  return { x: x, y: y };

}

  

// 绘制经纬度坐标

function drawCoordinates() {

  ctx.clearRect(0, 0, canvas.width, canvas.height); // 清空 Canvas

  

  ctx.fillStyle = "red";

  ctx.strokeStyle = "black";

  

  for (var i = 0; i < coordinates.length; i++) {

    var pixelCoord = convertLatLonToPixel(coordinates[i][0], coordinates[i][1]);

    ctx.beginPath();

    ctx.arc(pixelCoord.x, pixelCoord.y, 5, 0, Math.PI * 2);

    ctx.fill();

    ctx.stroke();

  }

  // 绘制连接线

  ctx.beginPath();

  ctx.moveTo(convertLatLonToPixel(coordinates[0][0], coordinates[0][1]).x, convertLatLonToPixel(coordinates[0][0], coordinates[0][1]).y);

  for (var i = 1; i < coordinates.length; i++) {

    var pixelCoord = convertLatLonToPixel(coordinates[i][0], coordinates[i][1]);

    ctx.lineTo(pixelCoord.x, pixelCoord.y);

  }

  ctx.stroke();

}

  

// 调用绘制函数

drawCoordinates();

</script>
